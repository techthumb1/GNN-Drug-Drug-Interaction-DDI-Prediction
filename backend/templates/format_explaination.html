def format_explanation_as_html(result, drug1_id, drug2_id): pred_score = result["prediction_score"] neighbors1 = result["top_neighbors_drug1"] neighbors2 = result["top_neighbors_drug2"] html = f"""

Prediction Score: {pred_score}

Top Neighbors for Drug {drug1_id}

{''.join(f'' for nid, score in neighbors1)}
Node IDSimilarity
{nid}{score}

Top Neighbors for Drug {drug2_id}

{''.join(f'' for nid, score in neighbors2)}
Node IDSimilarity
{nid}{score}
""" return html
